#### 控制 Y 轴 的显示位置（在左侧还是在右侧）

```js
option = {
```

#### 控制 Y 轴 的显示和隐藏
> 所有与 Y 轴有关的都会进行隐藏
>> 包括，Y轴 横向标示线（与X轴平行，且等长）
```js
option = {
    yAxis: [{
        // false 为隐藏，true 为默认值，展示
        show: false
    }]
};
```

#### 动态设置x-y轴上坐标的颜色(强制显示最后一个)
```js
const option = {

        type: 'category',
        data: data_x,
        // 设置坐标轴 刻度文字 展示的样式

            //    并不是 集合中的索引，而是当前是展示的第几个，隐藏的不算

                if(param === 1) return "#fff"
                else if(param === 2) return "red"
```

#### 动态设置x-y轴展示的刻度文字
- 只是展示文字发生改变，本来的功能没变
  - **比如，把100 换成了 200，但是 该刻度 表示的还是 100**
```js
const option = {
    xAxis: {
        type: 'category',
        data: x,
        axisLabel: {
            // 设置刻度展示的文字 
            formatter: function (value) {
                // 传入的是原始的 文字
                //    返回值是，展示的文字
                //    这样就可以 利用 条件判断，动态返回 数据
                if(value === 100) return 200
                return "xxx"
            }
        }
    },
    yAxis: {
        type: 'value',
        max: "dataMax",
        axisLabel: {
            // 设置刻度展示的文字 
            formatter: function (value) {
                // 传入的是原始的 文字
                //    返回值是，展示的文字
                //    这样就可以 利用 条件判断，动态返回 数据
                if(value === 100) return 200
                return "xxx"
            }
        }
    }
};
```



#### 设置x-y轴提示线
- 通常用在折线图上，用来标注一些重要的刻度

```js
const option = {
    xAxis: {
        type: 'category',
        data: x,
    },
    yAxis: {
        type: 'value',
    }
    series: [{
        data: y,
        type: 'line',

        // 设置提示线
        markLine: {
            // 去掉箭头
            symbol: ['none', 'none'],
            // 线的展示样式
            itemStyle: {
                normal: {
                    lineStyle: {
                        // 设置成虚线
                        type: 'dashed',
                        // 设置线的颜色
                        color: "red",
                        // 设置线的宽度
                        width: 4
                    },
                    label: {
                        show: false,
                        position:'middle'
                    }
                }
            },
            // 设置，在那个刻度上出现提示线
            //    一个属性就是一条 提示线
            //    x轴、y轴单独设置
            data: [{
                xAxis : x[ x.length - 1 ]
            }, {
                yAxis : y[ y.length - 1 ]
            }]
        }
    }]
};
```




#### 折线图—横轴注意点
1. 禁止出现刻度线，类似  `___|___|____|___|___`，就是这种竖着的东西
```js
const moren = {
    xAxis: {
        axisTick: {
            // 通过这个属性设置，true出现，false隐藏
            show: false,
        },
    }
}
```
2. 设置折线的拐点是位于，两个刻度之间，还是位于对应的刻度上
```js
const moren = {
    xAxis: {
        // 通过这个属性设置，true为中间，false为刻度上
        boundaryGap: false,
    }
}
```
3. 取消横轴坐标的自适应，全部显示出来
```js
const moren = {

            // 必须是数字0，字符串无效
            interval: 0
        },
    }
}
```

#### 饼图
> 创建一个简单的饼图
```js
// 构建饼图表
function createEcharts(dom, one1, one2, one3, one4) {
    var myChart = init(dom);
    const option = {
        tooltip: {
            trigger: 'item'
        },
        series: [
            {
                type: 'pie',
                // 饼图所占的比例
                radius: '76%',
                // 每一个块，对应的数据，最终会求出一个占比，用来绘制扇形
                data: [20, 30, 40, 50],
                // 扇形的夹脚小于多少，标签就不显示
                minShowLabelAngle: 0.001,
                // 如果标签重叠，默认隐藏，设置成下面，则重叠的标签，沿着Y轴往下移动
                labelLayout: {
                    moveOverlap: "shiftY"
                }
            },
        ]
    };
    myChart.setOption(option);
}
```

```js
function createEcharts(dom, one1, one2, one3, one4) {
    var myChart = init(dom);
    const option = {
        tooltip: {
            trigger: 'item'
        },
        series: [
            {
                type: 'pie',
                radius: '76%',
                // 饼图所占的比例
                radius: '76%',
                // 每一个块，对应的数据，最终会求出一个占比，用来绘制扇形
                data: [20, 30, 40, 50],
                // 扇形的夹脚小于多少，标签就不显示

                labelLayout: {
                    moveOverlap: "shiftY"
                }
                // 设置标签的样式
                label: {
                    // 出现在扇形中间，默认在外侧，用牵引线指着
                    position: "inside",
                    // 设置文字的样式
                    color: "#fff",
                    fontSize: 24
                }
            },
        ]
    };
    myChart.setOption(option);
}
```

```js
// 构建饼图表
function createEcharts(dom, one1, one2, one3, one4) {
    var myChart = init(dom);
    const option = {
        tooltip: {
            trigger: 'item'
        },
        series: [
            {
                type: 'pie',
                radius: '76%',
                // 单独设置每一块扇形区域的样式
                data: [
                    {   
                        // 对应的数据
                        value: one1,
                        // 扇形的颜色
                        itemStyle: {
                            color: '#FFC53D'
                        },
                        // 标签展示的文字
                        name: one1 + '%\r\n未接种',
                        // 标签的样式
                        label: {
                            // 扇形中间区域
                            position: "inner",
                            color: "#fff",
                            fontSize: 24
                        },
                    },
                    {
                        value: one4,
                        itemStyle: {
                            color: '#91D5FF'
                        },
                        name: one4 + '%\r\n＜18或＞75岁',
                        label: {
                            // 默认，在外侧
                            // position: "inner",
                            dx: -10,
                            color: "#fff",
                            fontSize: 24
                        }
                    },
                ]
            },
        ]
    };
    myChart.setOption(option);
}
```
